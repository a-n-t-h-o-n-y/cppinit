165  if (seperators_ == false) {

168 
169  // Insert seperator c, every n digits, right to left.
170  auto insert_seps = [](std::string& str, char c, std::size_t n) {
171  int count{0};
172  for (std::size_t i{str.size() - 1}; i < str.size(); --i) {
173  if ((count + 1) % n == 0 && i != 0) {
174  str.insert(i, 1, c);
175  }
176  ++count;
177  }
178  };
179 
180  if (rep_ == Representation::Binary) {
181  insert_seps(rep_str, ' ', 4);
182  } else if (rep_ == Representation::Hex) {
183  insert_seps(rep_str, ' ', 2);
184  } else if (rep_ == Representation::Decimal) {
